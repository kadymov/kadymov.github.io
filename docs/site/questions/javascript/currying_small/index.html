const memoizedCurry = (fn) => {
  const cache = new Map();
  return function curried(...args) {
    const key = args.join(',');
    if (cache.has(key)) return cache.get(key);

    const result = args.length >= fn.length 
      ? fn(...args) 
      : (...nextArgs) => curried(...args, ...nextArgs);

    cache.set(key, result);
    return result;
  };
};
