function GoodComponent({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    let isCancelled = false;

    fetch(`/api/users/${userId}`)
      .then(response => response.json())
      .then(userData => {
        if (!isCancelled) {
          setUser(userData);
        }
      });

    return () => {
      isCancelled = true;
    };
  }, [userId]);

  return <div>{user?.name}</div>;
}
